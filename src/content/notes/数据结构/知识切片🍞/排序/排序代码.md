## 冒泡排序

冒泡排序和快速排序统称为交换排序。

> 工具头文件：`utils.h`。存放一些多次使用的工具函数，便于代码复用。

```c
#include<stdio.h>

void swap(int *x, int *y) {
  int temp = *x;
  *x = *y;
  *y = temp;
}
int test_arr[] = {7,6,5,4,3,2,1};

void log_arr(int arr[], int n) {
  for (int i = 0; i < n; i ++ ) {
    printf("%d ", arr[i]);
  }
  printf("\n");
}
```

> `bubbleSort.c`。

```c
#include <stdio.h>
#include "utils.h"
enum {
  false,
  true
};

void bubbleSort(int array[], int n) {
  int left = 0, right = n - 1;
  for (int i = right; i > left; i --) { // 第几轮
    int flag = false;
    for (int j = left; j < i; j ++ ) {  // 每轮冒泡
      if (array[j] > array[j+1]) {
        swap(&array[j], &array[j+1]);
        flag = true;
      }
    }
    if (!flag) {
      return;
    }
  }
}

int main() {
  printf("冒泡排序开始！\n");
  int size = sizeof(test_arr) / sizeof(test_arr[0]);
  log_arr(test_arr, size);
  bubbleSort(test_arr, size);
  log_arr(test_arr, size);
  
  return 0;
}
```

## 快速排序

> `quickSort.c`。

```c
#include <stdio.h>
#include <stdlib.h>
#include "utils.h"

int Partition(int arr[], int l, int r) {
  // int pivotloc = (l+r) / 2;                       // 取中间位置，优化平均复杂度
  int pivotloc = rand() % (r - l + 1) + l;           // 取随机位置，优化平均复杂度
  swap(&arr[pivotloc], &arr[l]);
  int pivot = arr[l];
  while (l < r) {
    while (l < r && arr[r] >= pivot) r --;
    arr[l] = arr[r];
    while (l < r && arr[l] <= pivot) l ++;
    arr[r] = arr[l];
  }
  arr[l] = pivot;
  return l;
}

void quickSort(int arr[], int l, int r) {
  if (l < r) {
    int pivot_idx = Partition(arr, l, r);
    quickSort(arr, l, pivot_idx - 1);
    quickSort(arr, pivot_idx + 1, r);
  }
}

int main() {
  printf("快速排序开始！\n");
  int size = sizeof(test_arr) / sizeof(test_arr[0]);
  log_arr(test_arr, size);
  quickSort(test_arr, 0, size-1);
  log_arr(test_arr, size);
  return 0;
}
```


## 选择排序

> `selectSort.c`：简单选择排序。

```c
#include <stdio.h>
#include "utils.h"

void selectSort(int arr[], int n) {
  for (int i = 0; i < n - 1; i ++ ) {     // 进行n - 1趟，并且i为每次选择区间的最终交换位置
    int k = i;
    for (int j = i + 1; j < n; j ++ ) {
      if (arr[j] < arr[k]) {
        k = j;
      }
    } // 寻找当前区间内最小的数的下标
    swap(&arr[k], &arr[i]); // 与 i 位置（当前查找区间的第一个位置）进行交换
  }
}

int main() {
  printf("简单选择排序开始！\n");
  log_arr(test_arr, size);
  selectSort(test_arr, size);
  log_arr(test_arr, size);
  return 0;
}
```

> `heapSort.c`：堆排序。

```c
#include <stdio.h>
#include "utils.h"

int arr[] = {0, 7, 6, 5, 4, 3, 2, 1};

void DownAdjust(int a[], int k, int n)
{
  int i = k, j = i * 2; // i 为当前结点, j 为左孩子
  while (j <= n)
  {
    if (j + 1 <= n && a[j + 1] > a[j])
      j++; // 让 j 指向较大的结点
    if (a[i] < a[j])
    {
      swap(&a[i], &a[j]);
      i = j;
      j = 2 * i;
    }
    else
    {
      break;
    }
  }
}

void heapSort(int a[], int i, int n)
{
  // 建堆
  for (int i = n / 2; i >= 1; i--)
  {
    DownAdjust(a, i, n);
  }
  // 堆排序
  for (int i = n; i >= 2; i--)
  {
    swap(&a[i], &a[1]);
    DownAdjust(a, 1, i - 1);
  }
}

int main()
{
  int size = sizeof(arr) / sizeof(arr[0]);
  int n = size - 1;
  log_arr(arr, size);
  heapSort(arr, 1, n);
  log_arr(arr, size);
  return 0;
}
```
