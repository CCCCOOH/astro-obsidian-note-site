## 1. 完整实现

> 参考 [Hello算法](https://www.hello-algo.com/chapter_stack_and_queue/queue/#2)中的完整实现如下。

```c
#include <stdio.h>
#include <stdlib.h>
#include <assert.h>
typedef enum
{
  false,
  true
} bool;

typedef struct
{
  int *nums;       // 存放队列元素
  int front;       // 队首指针
  int queSize;     // 队尾指针，指向队尾 + 1
  int queCapacity; // 队列容量
} Queue;

Queue *newQueue(int capacity)
{
  Queue *que = (Queue *)malloc(sizeof(Queue));
  que->queCapacity = capacity;
  que->nums = (int *)malloc(sizeof(int) * capacity);
  que->front = que->queSize = 0;
  return que;
}

void del(Queue *que)
{
  // 通过指针分别释放队列结构体中的数组以及队列结构体本身
  free(que->nums);
  free(que);
}

bool empty(Queue *que)
{
  if (que->front == que->queSize)
  {
    printf("数组为空");
    return true;
  }
  printf("数组非空");
  return false;
}

int size(Queue *que)
{
  return que->queSize;
}

int peek(Queue *que)
{
  assert(size(que) != 0);   // 如果队列长度不为零才允许取出第一个元素
  return que->nums[que->front];
}

int capacity(Queue *que)
{
  return que->queCapacity;
}

void push(Queue *que, int num)
{
  // 如果长度和最大容量一样，显然队列已经爆满了！
  
  if (size(que) == capacity(que))
  {
    printf("队列已满\r\n");
    return ;
  }
  // rear = front + size
  int rear = (que->front+que->queSize) % que->queCapacity;
  que->nums[rear] = num;
  que->queSize ++;
}

int pop(Queue* que) 
{
  int num = peek(que);
  que->front = (que->front + 1) % que->queCapacity;
  que->queSize--;
  return num;
}

int *toArray(Queue* que, int *quesize)
{
  *quesize = que->queSize;
  int *res = (int *)calloc(que->queSize, sizeof(int)); // 自定分配并清零
  int j = que->front;
  for (int i = 0; i < que->queSize; i ++)
  {
    res[i] = que->nums[j%que->queCapacity];
    j++;
  }
  return res;
}

int main()
{
  Queue *queue = newQueue(10);
  push(queue, 2);push(queue, 5);push(queue, 3);
}
```

下面依次对各个模块进行拆解。

## 3. 构造函数

```c
typedef struct
{
  int *nums;       // 存放队列元素
  int front;       // 队首指针
  int queSize;     // 队尾指针，指向队尾 + 1
  int queCapacity; // 队列容量

```

类型定义+结构体，为队列维护以下属性：
- 数组（用于存储数据）
- 队首指针
- 队尾指针（`(front+queSize)%queCapacity` 公式实现，否则到达尾部就无法入队）
- 队列容量（用于计算队尾指针以及判满操作）

`Queue` 作为这里封装出的队列的类型名。

## 6. 判空操作

```c
bool empty(Queue *que)
{
  if (que->front == que->queSize)
  {
    printf("数组为空");
    return true;
  }
  printf("数组非空");
  return false;
}
```

判断队列是否为空，
- 如果队头和队尾所指的位置相同，则为空；
- 否则，队尾一直指向队尾元素+1，不为空。

## 8. 获取队首

```c

{
  assert(size(que) != 0);   // 如果队列长度不为零才允许取出第一个元素
  return que->nums[que->front];
}
```

- 如果队列有元素，即 `size(que)!=0`，返回队首元素。
- 否则，断言报错。

## 9. 入队

```c
void push(Queue *que, int num)
{
  // 如果长度和最大容量一样，显然队列已经爆满了！
  if (size(que) == capacity(que))
  {
    printf("队列已满\r\n");
    return ;
  }
  // rear = front + size
  int rear = (que->front+que->queSize) % que->queCapacity;
  que->nums[rear] = num;
  que->queSize ++;
}
```

- 如果队列爆满 `size = capacity` ，则不允许入队。
- 否则入队：
	- 计算队尾指针：`rear=(front+size)capacity`；
	- 因为队尾指针指向队尾+1，直接赋值。
	- 队列元素数量+1；

这里没有所谓的 “队尾指针”，队尾指针永远是用 `size` 实时计算出来的，所以增加 `size` 就相当于完成了队尾指针的后移逻辑。

## 10. 出队

```c
int pop(Queue* que) 
{
  int num = peek(que);
  que->front = (que->front + 1) % que->queCapacity;
  que->queSize--;
  return num;
}
```

- 弹出第一个元素并返回。
- 队头后移 `(front+1)%capacity`。
- 队列元素减少 `size-=1`。

## 11. 复制队列

由于这里相当于是一个环形的队列，所以将队列中的元素复制到一个数组中，便于打印操作。

```c
int *toArray(Queue* que, int *quesize)
{
  *quesize = que->queSize;
  int *res = (int *)calloc(que->queSize, sizeof(int)); // 自定分配并清零
  int j = que->front;
  for (int i = 0; i < que->queSize; i ++)
  {
    res[i] = que->nums[j%que->queCapacity];
    j++;
  }
  return res;
}
```

在C语言中必须要有一个变量来维护数组的长度，否则无法知道什么时候遍历结束。所以这里传参时要穿入一个 `queSize` 指针来更新 `size`，便于打印时知晓数组的长度。

- 用指针指向队头，
- 根据队列中元素的个数来进行遍历，
- 指针依次后移并对容量取模。

这里的 `calloc(size, typeSize)` 类似于 `malloc`，用于初始化 `size` 个大小为 `typeSize` 的空间。

## 12. 补充：独家朴素板子

Hello 算法的这个板子太长了，这里我提供一个更加简单的板子（自创）：

> 只实现了关键的几个操作。

```c
#include <stdio.h>
#include <stdlib.h>
#define rear (front+size)%capacity
const int capacity = 5;
int arr[capacity];

int front, size = 0;

void printArr()
{
  for (int i = 0; i < size; i ++)
  {
    printf("%d ", arr[(front+i)%capacity]);
  }
  puts("");
}

int pop()
{
  if (size <= 0)
  {
    puts("数组没元素，出什么队呢？");
    return -1;
  }
  size--;
  printArr();
  return arr[front];
}

void push(int elem)
{
  if (size >= capacity)
  {
    puts("队列爆满了！");
    return;
  }
  arr[rear] = elem;
  size++;
  printArr();
}

// 测试队列效果
int main(void) 
{
  push(1);
  push(2);
  push(3);
  push(4);
  push(5);
  pop();
  pop();
  push(23);
  push(14);
  pop();
  pop();
  pop();
  pop();
}
```
